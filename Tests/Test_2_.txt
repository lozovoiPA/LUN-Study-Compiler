real i = 0;
while (i < n) 
{ 
    read(array[i]);

real is_swapped = 1;
while (is_swapped == 1) 
{
    is_swapped = 0;
    i = 0;
    while (i < n - 1) 
    { 
        if (array[i] > array[i + 1]) 
        {
            real temp = array[i];
            array[i] = array[i + 1]; 
            array[i + 1] = temp; 
            is_swapped = 1; 
        }
        i = i + 1;
    }
}
